					end, 100, 0);
				end
			elseif panelState == 3 then
				if selectedDataBox[3][1] then
					userData[3][1] = string.sub(userData[3][1], 0, string.len(userData[3][1])-1);
					delTimer = setTimer(function()

					end, 100, 0);
				end
			end
